fetch('bilder/galerie/gallery.json') .then(response => response.json()) .then(data => { const gallery = document.getElementById('gallery'); Object.keys(data).sort().reverse().forEach(folder => { const title = document.createElement('h2'); title.textContent = folder; gallery.appendChild(title); data[folder].forEach(img => { const image = document.createElement('img'); image.src = `bilder/galerie/${folder}/${img}`; gallery.appendChild(image); }); }); });
