fetch("bilder/galerie/gallery.json") .then(res => res.json()) .then(data => { const galerie = document.getElementById("galerie"); Object.entries(data).forEach(([monat, bilder]) => { const h2 = document.createElement("h2"); h2.textContent = monat; galerie.appendChild(h2); bilder.forEach(datei => { const img = document.createElement("img"); img.src = `bilder/galerie/${monat}/${datei}`; img.alt = datei; img.style.maxWidth = "250px"; img.style.margin = "10px"; galerie.appendChild(img); }); }); });
